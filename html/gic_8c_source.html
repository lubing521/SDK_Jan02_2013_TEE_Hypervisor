194 int gic_dist_init(void)
195 {
196  u32 gic_nr = 0;
197  va_t base = GIC_DIST;
198  u32 irq_start = IRQ_GIC_START;
199  unsigned int max_irq, i, it_lines_number;
200 
201  /* Current processor id */
202  /* It is the primary processor */
203  u32 cpumask = 1 << get_cpu_id();
204 
205  if (GIC_MAX_NR <= gic_nr) {
206  return OTZ_EFAIL;
207  }
208 
209 
210  cpumask |= cpumask << 8;
211  cpumask |= cpumask << 16;
212 
213  gic_data[gic_nr].dist_base = base;
214  gic_data[gic_nr].irq_offset = (irq_start - 1) & ~31;
215 
216  gic_write(0, base + GIC_DIST_CTRL);
217 
218  /*
219  * Find out how many interrupts are supported.
220  */
221  max_irq = gic_read(base + GIC_DIST_CTR) & 0x1f;
222  it_lines_number = max_irq;
223 
224  max_irq = (max_irq + 1) * 32;
225 
226  /*
227  * The GIC only supports up to 1020 interrupt sources.
228  * Limit this to either the architected maximum, or the
229  * platform maximum.
230  */
231  if (max_irq > max(1020, GIC_NR_IRQS))
232  max_irq = max(1020, GIC_NR_IRQS);
233 
234 /* Set ALL interrupts as non-secure interrupts */
235 #if 1
236 
237 
238  for(i = 1; i <= it_lines_number; i++) {
239  gic_write(0xffffffff,
240  base + GIC_DIST_SECURITY + i * 4);
241 #ifndef CONFIG_KSPACE_PROTECTION
242  sw_printf("SW: gic security register 0x%x and value %x\n",
243  i, gic_read(base + GIC_DIST_SECURITY + i *4));
244 #endif
245  }
246 #endif
247 #ifndef TIMER_NOT_DEFINED
248 #ifndef CONFIG_EMULATE_FIQ
249 {
250 /* Modify the Free Running Timer Interrupt used by secure kernel
251  * as Group 0 interrupt */
252  u32 mask;
253  u32 gic_irq;
254 
255  gic_irq = FREE_RUNNING_TIMER_IRQ - gic_data[gic_nr].irq_offset;
256 
257  mask = gic_read(base + GIC_DIST_SECURITY + (gic_irq / 32) * 4);
258 
259  mask &= (~(1 << (FREE_RUNNING_TIMER_IRQ % 32)) & 0xffffffff);
260  gic_write(mask, base +
261  GIC_DIST_SECURITY + (gic_irq / 32) * 4);
262 
263 /* Modify the Tick Timer Interrupt used by secure kernel
264  * as Group 0 interrupt */
265  gic_irq = TICK_TIMER_IRQ - gic_data[gic_nr].irq_offset;
266 
267  mask = gic_read(base + GIC_DIST_SECURITY + (gic_irq / 32) * 4);
268 
269  mask &= (~(1 << (TICK_TIMER_IRQ % 32)) & 0xffffffff);
270  gic_write(mask, base +
271  GIC_DIST_SECURITY + (gic_irq / 32) * 4);
272 }
273 #endif
274 #endif
275  /*
276  * Set all global interrupts to be level triggered, active low.
277  */
278  for (i = 32; i < max_irq; i += 16)
279  gic_write(0, base + GIC_DIST_CONFIG + i * 4 / 16);
280 
281  /*
282  * Set all global interrupts to this CPU only.
283  */
284  for (i = 32; i < max_irq; i += 4)
285  gic_write(cpumask, base + GIC_DIST_TARGET + i * 4 / 4);
286 
287  /*
288  * Set priority on all interrupts.
289  */
290  for (i = 0; i < max_irq; i += 4)
291  gic_write(0xa0a0a0a0, base + GIC_DIST_PRI + i * 4 / 4);
292 
293  /*
294  * Disable all interrupts.
295  */
296  for (i = 32; i < max_irq; i += 32)
297  gic_write(0xffffffff,
298  base + GIC_DIST_ENABLE_CLEAR + i * 4 / 32);
299 
300  /* Enable both group0 and group1 interrupts */
301  gic_write(0x3, base + GIC_DIST_CTRL);
302 
303  return OTZ_OK;
304 }

311 int gic_cpu_init(void)
312 {
313  u32 gic_nr = 0;
314  int cpuid = get_cpu_id();
315  va_t base = GIC_CPU + GIC_BANK_OFFSET * cpuid;
316  va_t dist_base = GIC_DIST + GIC_BANK_OFFSET * cpuid;
317 
318  if (GIC_MAX_NR <= gic_nr) {
319  return OTZ_EFAIL;
320  }
321 
322  gic_data[gic_nr].cpu_base = base;
323 
324 #ifndef CONFIG_KSPACE_PROTECTION
325  sw_printf("SW: gic cpu base 0x%x\n", gic_data[gic_nr].cpu_base);
326 #endif
327 
328  /* Set the Per-CPU interrupts 15-7 as Secure and the rest
329  * as Non-secure */
330  gic_write(0xffff00ff,
331  dist_base + GIC_DIST_SECURITY);
332 
333  /* Enable all SGIs for the CPU */
334  gic_write(0x0000ffff, dist_base + GIC_DIST_ENABLE_SET);
335  /* Disable all PPIs for the CPU */
336  gic_write(0xffff0000, dist_base + GIC_DIST_ENABLE_CLEAR);
337 
338 /* Set priority to higher or equal than 0x80 for non-secure access */
339  gic_write(0xf0, base + GIC_CPU_PRIMASK);
340 /* Select ackctl, IRQ enable and Group0 and Group1 interrupts */
341 #ifndef CONFIG_EMULATE_FIQ
342  gic_write(0xb, base + GIC_CPU_CTRL);
343 #else
344  gic_write(0xf, base + GIC_CPU_CTRL);
345 #endif
346 
347  return OTZ_OK;
348 }
