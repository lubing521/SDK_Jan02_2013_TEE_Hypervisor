41 s32int sw_sleep(u32 seconds)
42 {
43  u64 current_ticks = 0,end_value = (seconds*10);
44  /* we have a 1mhz clock. The current period of interrupts is one every
45  * 100ms. We need to wait till we get seconds*10 interrupts.
46  */
47  current_ticks = get_current_ticks();
48  while((get_current_ticks() - current_ticks) < end_value);
49  return(0);
50 }
