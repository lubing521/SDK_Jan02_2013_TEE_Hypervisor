165 s32int check_space_for_object(u8* str, s32int width, s32int height)
166 {
167  if((obj_val.obj_x + width) > obj_val.window_x2) {
168  obj_val.max_height = obj_val.curr_height;
169  obj_val.curr_height = obj_val.obj_height;
170  obj_val.obj_x = obj_val.window_x1+10;
171  obj_val.obj_y += obj_val.max_height + obj_val.space_bw_obj;
172  if(obj_val.obj_y > obj_val.window_y2) {
173  obj_val.end_inserting_obj = 1;
174  return 0;
175  }
176  }
177  return 1;
178 }
