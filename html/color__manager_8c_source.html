63 u32 calc_contrast_color (u32 input_color){
64 
65  if(abs(((input_color ) & color(BLUE)) - color(NAVY)) <=
66  color(DARK_BLUE) &&
67  abs(((input_color >> 8) & color(BLUE)) - color(NAVY)) <=
68  color(DARK_BLUE) &&
69  abs(((input_color >> 16) & color(BLUE)) - color(NAVY)) <=
70  color(DARK_BLUE))
71  return (color(GRAY) + input_color) & color(WHITE);
72 
73  else return input_color ^ color(WHITE);
74 }
