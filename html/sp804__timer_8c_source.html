216 void secure_timer_init(u32 timer_base)
217 {
218  u32 scctrl;
219  int init_done = FALSE;
220  /* Select 1MHz TIMCLK as the reference clock for SP804 timers */
221  scctrl = sw_readl((void *)(SYSCTL_BASE + SCCTRL));
222 
223  if(timer_base == TIMER0_BASE)
224  init_done = (scctrl & SCCTRL_TIMEREN0SEL_TIMCLK) ? 1 : 0;
225  else if(timer_base == TIMER1_BASE)
226  init_done = scctrl & SCCTRL_TIMEREN1SEL_TIMCLK ? 1: 0 ;
227  else if(timer_base == TIMER2_BASE)
228  init_done = scctrl & SCCTRL_TIMEREN2SEL_TIMCLK ? 1: 0 ;
229  else if(timer_base == TIMER3_BASE)
230  init_done = scctrl & SCCTRL_TIMEREN3SEL_TIMCLK ? 1: 0 ;
231 
232  /* If already a clock is selected for the Timer then return */
233  if(init_done)
234  return;
235 
236  if(timer_base == TIMER0_BASE)
237  scctrl |= SCCTRL_TIMEREN0SEL_TIMCLK;
238  else if(timer_base == TIMER1_BASE)
239  scctrl |= SCCTRL_TIMEREN1SEL_TIMCLK;
240  else if(timer_base == TIMER2_BASE)
241  scctrl |= SCCTRL_TIMEREN2SEL_TIMCLK;
242  else if(timer_base == TIMER3_BASE)
243  scctrl |= SCCTRL_TIMEREN3SEL_TIMCLK;
244 
245  sw_writel(scctrl, (void*)(SYSCTL_BASE + SCCTRL));
246 }

282 void secure_set_timer(u32 timer_base, u32 tim_mod_index, u32 mode, u32 usecs)
283 {
284  u32 scctrl, val, reg_offset;
285  int sel_clk = FALSE;
286 
287  scctrl = sw_readl((void *)(SYSCTL_BASE + SCCTRL));
288 
289  if(timer_base == TIMER0_BASE)
290  sel_clk = (scctrl & SCCTRL_TIMEREN0SEL_TIMCLK) ? 1 : 0;
291  else if(timer_base == TIMER1_BASE)
292  sel_clk = scctrl & SCCTRL_TIMEREN1SEL_TIMCLK ? 1: 0 ;
293  else if(timer_base == TIMER2_BASE)
294  sel_clk = scctrl & SCCTRL_TIMEREN2SEL_TIMCLK ? 1: 0 ;
295  else if(timer_base == TIMER3_BASE)
296  sel_clk = scctrl & SCCTRL_TIMEREN3SEL_TIMCLK ? 1: 0 ;
297 
298  if(!sel_clk)
299  return;
300 
301  switch(tim_mod_index){
302  case 0:
303  reg_offset = 0x0;
304  break;
305  case 1:
306  reg_offset = 0x20;
307  break;
308  default:
309  reg_offset = 0;
310  }
311 
312  /* Setup Timer for generating irq */
313  val = sw_readl((void *)(timer_base + TIMER_CTRL + reg_offset));
314  val &= ~TIMER_CTRL_ENABLE;
315  val |= (TIMER_CTRL_32BIT | TIMER_CTRL_IE);
316 
317  if(mode == MODE_FREE_RUNNING) {
318  val &= ~TIMER_CTRL_ONESHOT;
319  val &= ~TIMER_CTRL_PERIODIC;
320  }
321  else if(mode == MODE_ONESHOT) {
322  val |= TIMER_CTRL_ONESHOT;
323  }
324  else{ /* Mode is Periodic */
325  val &= ~TIMER_CTRL_ONESHOT;
326  val |= TIMER_CTRL_PERIODIC;
327  }
328 
329  sw_writel(val, (void *)(timer_base + TIMER_CTRL + reg_offset ));
330  sw_writel(usecs, (void *)(timer_base + TIMER_LOAD + reg_offset));
331 }
