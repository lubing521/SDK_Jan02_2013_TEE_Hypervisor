129 void serial_init(u32 uart_id)
130 {
131  int lcrh_reg;
132  /* First, disable everything */
133  write_uart(UART_PL011_CR, 0x0, uart_id);
134 
135  lcrh_reg = read_uart(UART_PL011_LCRH, uart_id);
136  lcrh_reg &= ~UART_PL011_LCRH_FEN;
137  write_uart(UART_PL011_LCRH, lcrh_reg, uart_id);
138 
139  /*
140  * Set baud rate
141  *
142  baudrate = 115200
143  UART clock = 24000000
144  * IBRD = UART_CLK / (16 * BAUD_RATE)
145  * FBRD = RND((64 * MOD(UART_CLK,(16 * BAUD_RATE)))
146  * / (16 * BAUD_RATE))
147  */
148 
149  write_uart(UART_PL011_IBRD, 13, uart_id);
150 
151  /* Set the UART to be 8 bits, 1 stop bit,
152  * no parity, fifo enabled
153  */
154  write_uart(UART_PL011_LCRH, (UART_PL011_LCRH_WLEN_8 | UART_PL011_LCRH_FEN), uart_id);
155 
156  /* Finally, enable the UART */
157  write_uart(UART_PL011_CR, (UART_PL011_CR_UARTEN | UART_PL011_CR_TXE), uart_id);
158 
159  return;
160 }
