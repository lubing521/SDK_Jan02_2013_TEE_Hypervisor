67 int tzhyp_guest_init(void)
68 {

80 
81 #ifdef CONFIG_MULTI_GUESTS_SUPPORT

85  ns_world_binfo[guest].nskbi_srcaddr = (u32)&kernel_2_start;

87  NSK_LOAD_ADDRESS + GUEST_MEM_SIZE;

91 #ifndef CONFIG_ZYNQ7_BOARD

100

103  /* Load all the guests to memory */
