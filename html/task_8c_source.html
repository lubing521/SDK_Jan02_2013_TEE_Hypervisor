159  struct list *l;
160  bool found;

162  struct sw_global *pglobal = &global_val;

171  found = FALSE;
172  list_for_each(l, &pglobal->task_list) {

175  found = TRUE;

178  }
179

182  }
183 
184  list_del(&task->head);
185 
186  if (!list_empty(&pglobal->ready_to_run_list)) {
187  ready_task= NULL;
188  list_for_each(l, &pglobal->ready_to_run_list) {
189  ready_task = list_entry(l, struct sw_task, ready_head);
190  if (ready_task->task_id == task_id) {
191  list_del(&ready_task->ready_head);
192  break;
193  }
194  }
195  }
196 
197  cpu_task_exit(task);
198 
199  if(task->tls)
200  sw_free_private(task_id, task->tls);
201  sw_free_private(task_id, task->task_sp);
202  sw_free_private(task_id, task);
203  if(free_private_heap(task_id) == -1){
204  sw_printf("SW: Cannot free this heap\n");
205  }
206  return OTZ_OK;
207 }
