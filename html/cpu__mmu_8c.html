{
/* Domain is a two bit field
00 = no access,
01=client,
10=reserved,
11=manager
*/
u32 value;
value = read_domain();
//clear the current domain
u32 mask = ~(0b11 << (domain*2));
value = value & mask;
//Set the domain
value = value | (access << ((domain)*2));
write_domain(value);
}
