411 void get_api_context(int *svc_id, int *task_id, int *cmd_id, int *cmd_type)
412 {
413  pa_t cmd_phy;
414  struct otz_smc_cmd *cmd = NULL;
415 
416  *svc_id = OTZ_SVC_INVALID;
417  *task_id = 0;
418  *cmd_id = OTZ_GLOBAL_CMD_ID_INVALID;
419  *cmd_type = params_stack[2];
420 
421  if(*cmd_type == OTZ_CMD_TYPE_NS_TO_SECURE) {
422  cmd_phy = (pa_t) params_stack[1];
423  if(!cmd) {
424  if(map_to_ns(cmd_phy, (va_t*) &cmd))
425  return;
426  }
427  }
428  else {
429  cmd = (struct otz_smc_cmd *)params_stack[1];
430  }
431 
432 #ifdef SECURE_API_DBG
433  sw_printf("SW: get_api_context: cmd va 0x%x cmd pa 0x%x\n",
434  cmd, va_to_pa_ns(cmd));

437  cmd, va_to_pa(cmd));
438 #endif
439 
440  *svc_id = ((cmd->id >> 10) & (0x3ff));
441  *cmd_id = (cmd->id & 0x3ff);
442  *task_id = cmd->context;

473 int decode_data(void *data,
474  struct otzc_encode_meta *meta_data,
475  int *type, int *offset, int *pos,
476  int *mapped, void **out_data, int *out_len)
477 {
478  int ret = 0;
479  int temp_offset = *offset, temp_pos = *pos;
480  *mapped = 0;
481 
482  switch(meta_data[temp_pos].type) {
483  case OTZ_ENC_UINT32: {
484  *offset = temp_offset + sizeof(u32);
485  *pos = temp_pos + 1;
486  *type = OTZ_ENC_UINT32;
487  *out_data = data + temp_offset;
488  *out_len = meta_data[temp_pos].len;
489  break;
490  }
491  case OTZ_ENC_ARRAY: {
492  *offset = temp_offset + meta_data[temp_pos].len;
493  *pos = temp_pos + 1;
494  *type = OTZ_ENC_ARRAY;
495  *out_data = data + temp_offset;
496  *out_len = meta_data[temp_pos].len;
497  break;
498  }
499  case OTZ_MEM_REF: {
500  if(map_to_ns((pa_t)(*((u32*)data + temp_offset)), (va_t*)out_data) != 0)
501  return -1;
502  *offset = temp_offset + sizeof(u32);
503  *type = OTZ_MEM_REF;
504  *pos = temp_pos + 1;
505  *mapped = 1;
506  *out_len = meta_data[temp_pos].len;
507  break;
508  }
509  case OTZ_SECURE_MEM_REF: {
510  *out_data = (void*)(*((u32*)data + temp_offset));
511  *offset = temp_offset + sizeof(u32);
512  *type = OTZ_MEM_REF;
513  *pos = temp_pos + 1;
514  *mapped = 0;
515  *out_len = meta_data[temp_pos].len;
516  break;
517  }
518  default:
519  ret = -1;
520  break;
521  }
522  return ret;
523 }

702

706  (OTZ_MAX_RES_PARAMS + OTZ_MAX_REQ_PARAMS ))) {

708  enc_context->enc_res_offset)
