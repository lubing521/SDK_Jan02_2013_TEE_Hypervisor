499 int process_otz_crypto_svc(u32 svc_cmd_id, void *req_buf, u32 req_buf_len,
500  void *resp_buf, u32 res_buf_len,
501  struct otzc_encode_meta *meta_data, u32 *ret_res_buf_len)
502 {
503  int ret_val = SMC_ERROR;
504 
505  if((req_buf_len == 0) || (res_buf_len == 0) ) {
506  return(0);
507  }
508 
509  switch (svc_cmd_id) {
510  case OTZ_CRYPT_CMD_ID_LOAD_LIBS:
511  process_otz_crypto_load_libs();
512  break;
513  case OTZ_CRYPT_CMD_ID_UNLOAD_LIBS:
514  process_otz_crypto_unload_libs();
515  break;
516  case OTZ_CRYPT_CMD_ID_ENCRYPT:
517  case OTZ_CRYPT_CMD_ID_DECRYPT:
518  break;
519  case OTZ_CRYPT_CMD_ID_MD5:
520  case OTZ_CRYPT_CMD_ID_SHA1:
521  case OTZ_CRYPT_CMD_ID_SHA224:
522  case OTZ_CRYPT_CMD_ID_SHA256:
523  case OTZ_CRYPT_CMD_ID_SHA384:
524  case OTZ_CRYPT_CMD_ID_SHA512:
525  ret_val = process_otz_crypto_digest_cmd(req_buf,req_buf_len,
526  resp_buf,res_buf_len,meta_data,
527  ret_res_buf_len,svc_cmd_id);
528  break;
529  case OTZ_CRYPT_CMD_ID_HMAC_MD5:
530  case OTZ_CRYPT_CMD_ID_HMAC_SHA1:
531  case OTZ_CRYPT_CMD_ID_HMAC_SHA224:
532  case OTZ_CRYPT_CMD_ID_HMAC_SHA256:
533  case OTZ_CRYPT_CMD_ID_HMAC_SHA384:
534  case OTZ_CRYPT_CMD_ID_HMAC_SHA512:
535  ret_val = process_otz_crypto_hmac_cmd(req_buf,req_buf_len,
536  resp_buf,res_buf_len,meta_data,
537  ret_res_buf_len,svc_cmd_id);
538  break;
539  case OTZ_CRYPT_CMD_ID_CIPHER_AES_128_CBC:
540  case OTZ_CRYPT_CMD_ID_CIPHER_AES_128_ECB:
541  case OTZ_CRYPT_CMD_ID_CIPHER_AES_128_CTR:
542  case OTZ_CRYPT_CMD_ID_CIPHER_AES_128_XTS:
543  case OTZ_CRYPT_CMD_ID_CIPHER_DES_ECB:
544  case OTZ_CRYPT_CMD_ID_CIPHER_DES_CBC:
545  case OTZ_CRYPT_CMD_ID_CIPHER_DES3_ECB:
546  case OTZ_CRYPT_CMD_ID_CIPHER_DES3_CBC:
547  ret_val = process_otz_crypto_cipher_cmd(req_buf,req_buf_len,
548  resp_buf,res_buf_len,meta_data,
549  ret_res_buf_len,svc_cmd_id);
550  break;
551  default:
552  ret_val = SMC_EOPNOTSUPP;
553  break;
554  }
555  return ret_val;
556 }
