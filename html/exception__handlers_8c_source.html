137 void print_data_abort()
138 {
139  u32 dfsr = read_dfsr();
140  u32 dfar = read_dfar();
141  u32 fault_status = (dfsr & 0xF) | ((dfsr & 0x400) >> 6);
142 
143  sw_printk("SW: Data Abort Address: %08x\n", dfar);
144  sw_printk("SW: Fault type: ");
145  sw_printk((char*)data_abt_fault_string[fault_status]);
146  sw_printk("SW: (%x), domain %x, Write not Read: %x, External: %x\n",
147  fault_status, ((dfsr & FSR_DOMAIN_MASK) >> 4), ((dfsr & FSR_WNR_MASK) >> 11), ((dfsr & FSR_EXT_MASK) >> 12 ));
148 }
