206 struct sw_task* get_next_task(void)
207 {
208  struct list *l;
209  struct sw_task *current_task = get_current_task();
210  struct sw_task *next_task;
211  struct sw_task *temp_task;
212  irq_flags_t irqflags;
213 
214  spin_lock_irqsave(&global_val.ready_to_run_lock, &irqflags);
215 
216  if (list_empty(&global_val.ready_to_run_list)) {
217  if(current_task != NULL && current_task->state != TASK_STATE_WAIT &&
218  current_task->state != TASK_STATE_SUSPEND){
219  next_task = current_task;
220  goto func_return;
221  }
222  else{
223  next_task = NULL;
224  goto func_return;
225  }
226  }
227 
228  l = list_pop(&global_val.ready_to_run_list);
229  next_task = list_entry(l, struct sw_task, ready_head);
230 
231  list_for_each_entry(temp_task, &global_val.ready_to_run_list ,ready_head){
232  if(temp_task == current_task)
233  goto func_return;
234  }
235 
236  if(current_task != NULL && current_task->state != TASK_STATE_WAIT &&
237  current_task->state != TASK_STATE_SUSPEND) {
238  schedule_task(current_task);
239  }
240 
241 func_return:
242  spin_unlock_irqrestore(&global_val.ready_to_run_lock,irqflags);
243  return next_task;
244 }
