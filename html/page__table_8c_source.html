530 int map_section_entry(u32 *pgd, struct cpu_section *entry)
531 {
532  u32 l1_index = 0;
533  u32 *l1_pte;
534 
535  l1_index = entry->va >> 20;
536  l1_pte = pgd + l1_index;
537 
538  switch ((*l1_pte) & L1_TYPE_MASK) {
539  case (L1_TYPE_FAULT):
540  *l1_pte = ( entry->pa & SECTION_BASE_MASK) | L1_TYPE_SECTION;
541  *l1_pte |= (entry->dom << L1_DOMAIN_SHIFT);
542  *l1_pte |= (entry->b << L1_SECTION_B_SHIFT);
543  *l1_pte |= (entry->c << L1_SECTION_C_SHIFT);
544  *l1_pte |= (entry->tex << L1_SECTION_TEX_SHIFT);
545  *l1_pte |= (entry->xn << L1_SECTION_XN_SHIFT);
546  *l1_pte |= ((entry->ap & L1_SECTION_AP_MASK) << L1_SECTION_AP_SHIFT);
547  *l1_pte |= (((entry->ap >> 2) & 0x1) << L1_SECTION_AP2_SHIFT);
548  *l1_pte |= (entry->s << L1_SECTION_S_SHIFT);
549  *l1_pte |= (entry->ng << L1_SECTION_NG_SHIFT);
550  *l1_pte |= (entry->ns << L1_SECTION_NS_SHIFT);
551 
552 /*
553  flush is not needed - page table is cacheable
554 */
555  break;
556  case (L1_TYPE_SECTION):
557 /*
558  Change in flags need to be handled
559 */
560  break;
561  default:
562  sw_printk("SW: section_map_hva_to_hpa: This function supports "
563  "only section mappings.\n");
564  sw_printk("SW: L1 descriptor: %08x\n", *l1_pte);
565  return -1;
566  }
567 
568  return 0;
569 }

761 int page_ref_add(u32 l1_pte)
762 {
763  struct list *l;
764  struct sw_global *pglobal = &global_val;
765  struct sw_page_ref *page_ref = NULL;
766  bool found = FALSE;
767 
768  if (!list_empty(&pglobal->page_ref_list)) {
769  list_for_each(l, &pglobal->page_ref_list) {
770  page_ref = list_entry(l, struct sw_page_ref, head);
771  if (page_ref->l1_pte == l1_pte) {
772  found = TRUE;
773  break;
774  }
775  }
776  }
777 
778  if (!found) {
779  page_ref = (struct sw_page_ref*)sw_malloc_private(COMMON_HEAP_ID,
780  sizeof(struct sw_page_ref));
781  if(!page_ref) {
782  sw_printf("SW: page reference allocation: malloc failed\n");
783  return OTZ_EFAIL;
784  }
785  page_ref->l1_pte = l1_pte;
786  page_ref->ref_cnt = 0;
787  list_add_tail(&pglobal->page_ref_list, &page_ref->head);
788  }
789 
790  page_ref->ref_cnt++;
791 #ifdef PAGETABLE_DBG
792  sw_printf("SW: add refer count 0x%x and l1_pte 0x%x\n", page_ref->ref_cnt, page_ref->l1_pte);
793 #endif
794  return OTZ_OK;
795 }

809 {

820  found = TRUE;
821  break;

825
