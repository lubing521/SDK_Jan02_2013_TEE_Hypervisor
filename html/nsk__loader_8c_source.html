47 int nsk_load(struct nsk_boot_info *ni)
48 {
49  int i, error;
50  u32 *addr;
51  u32 ns_map_size, s_map_size, aligned_size;
52 
53  sw_printf("nsk load:");
54 
55  aligned_size = ni->nskbi_size;
56  /* Align it to the next 1MB */
57  if (aligned_size & (MAP_SIZE - 1)) {
58  aligned_size = (aligned_size & ALIGN_MASK) + MAP_SIZE;
59  }
60 
61  ns_map_size = aligned_size;
62  if (ni->nskbi_loadaddr & (MAP_SIZE - 1)) {
63  ns_map_size += MAP_SIZE;
64  }
65 
66  /* map the non secure area */
67  error = map_nsmemsect_normal(ni->nskbi_loadaddr & ALIGN_MASK,
68  ni->nskbi_loadaddr & ALIGN_MASK,
69  ns_map_size);
70  if (error < 0) {
71  sw_printf("nsk_load: mapping failed\n");
72  return error;
73  }
74 
75  s_map_size = aligned_size;
76  /* start address is assumed to be aligned */
77  error = map_secure_memory(ni->nskbi_srcaddr, ni->nskbi_srcaddr,
78  s_map_size, PTF_PROT_KRW);
79  if (error < 0) {
80  sw_printf("nsk_load: mapping failed\n");
81  return error;
82  }
83 
84  sw_printf("load addr = %x, startaddr = %x, size = %x\n",
85  ni->nskbi_loadaddr, ni->nskbi_srcaddr, ni->nskbi_size);
86 
87  /* Move the ns kernel to the load address */
88  sw_memcpy((void*)ni->nskbi_loadaddr, (void*)ni->nskbi_srcaddr,
89  ni->nskbi_size);
90 
91  dsb();
92 
93  flush_cache_all();
94 
95  /* Unmap the non secure area */
96  unmap_nsmemsect_normal(ni->nskbi_loadaddr, ns_map_size);
97 
98  unmap_secure_memory(ni->nskbi_srcaddr, s_map_size);
99 
100  /* Load initrd */
101  if (ni->nskbi_initrd_flag) {
102  nsk_initrd_load(ni);
103  }
104 
105  sw_printf("Done\n");
106
