119 int buddy_init(int index, void *heap_start, unsigned int heap_size)
120 {
121  int cntr = 0, tnodes = 0;
122  void * mem_start;
123  unsigned int mem_size, hk_total_size;
124  struct sw_free_area * freenode = NULL;
125 
126  /* We manage heap space only in power of two */
127  if (!IS_POW_TWO(heap_size)) {
128  return OTZ_EFAIL;
129  }
130 
131  hk_total_size = (heap_size * HOUSE_KEEPING_PERCENT) / 100;
132 
133  /* keep 4K of initial heap area for our housekeeping */
134  buddy_heap[index].hk_fn_array = (struct sw_free_area *)heap_start;
135  buddy_heap[index].hk_fn_count = (hk_total_size / 2) / sizeof(struct sw_free_area);
136  buddy_heap[index].hk_an_array =
137  (struct sw_alloced_area *)(heap_start + (hk_total_size / 2));
138  buddy_heap[index].hk_an_count = (hk_total_size / 2) /
139  sizeof(struct sw_alloced_area);
140  sw_memset(buddy_heap[index].hk_fn_array, 0, hk_total_size);
141 
142  INIT_LIST_HEAD(&buddy_heap[index].current.head);
143 
144  mem_start = heap_start + hk_total_size;
145  mem_size = heap_size - hk_total_size;
146  buddy_heap[index].mem_start = mem_start;
147  buddy_heap[index].mem_size = mem_size;
148  buddy_heap[index].heap_start = heap_start;
149  buddy_heap[index].heap_size = heap_size;
150  sw_memset(&buddy_heap[index].free_area[0], 0, sizeof(buddy_heap[index].free_area));
151  for (cntr = 0; cntr < BINS_MAX_ORDER; cntr++) {
152  INIT_LIST_HEAD(&buddy_heap[index].free_area[cntr].head);
153  }
154 
155  INIT_LIST_HEAD(&buddy_heap[index].free_area[BINS_MAX_ORDER - 1].head);
156 
157  while (mem_size > 0) {
158  freenode = get_free_hk_node(index);
159  if (!freenode) {
160  OTZ_DPRINTK("no free node\n");
161  return -1;
162  }
163  freenode->map = mem_start;
164  mem_size -= MAX_BLOCK_SIZE;
165  mem_start += MAX_BLOCK_SIZE;
166  list_add_tail(&buddy_heap[index].free_area[BINS_MAX_ORDER - 1].head,
167  &freenode->head);
168  buddy_heap[index].free_area[BINS_MAX_ORDER - 1].count++;
169  tnodes++;
170 
171  }
172  OTZ_DPRINTK("Total: 0x%x nodes of size 0x%x added to last bin.\n",
173  tnodes, MAX_BLOCK_SIZE);
174 
175  return 0;
176 }

186 static struct sw_free_area *buddy_get_contiguous_block(int index, unsigned int num_blocks,
187  unsigned int idx)
188 {
189  struct sw_free_area *snode = NULL, *cnode = NULL, *pnode = NULL;
190  struct list *cnhead;
191  unsigned int count = 0;
192 
193  if (idx >= BINS_MAX_ORDER) {
194  return NULL;
195  }
196 
197  if (list_empty(&buddy_heap[index].free_area[idx].head)) {
198  return NULL;
199  }
200 
201  /* First check if we have enough nodes, contiguous or non-contiguous. */
202  if (buddy_heap[index].free_area[idx].count >= num_blocks) {
203  /* okay we have enough nodes. Now try allocation contiguous nodes */
204  list_for_each(cnhead, &buddy_heap[index].free_area[idx].head) {
205  cnode = list_entry(cnhead, struct sw_free_area, head);
206  if (cnode) {
207  if (snode == NULL) {
208  snode = cnode;
209  }
210  if (pnode) {
211  if (pnode->map + MAX_BLOCK_SIZE ==
212  cnode->map) {
213  pnode = cnode;
214  if (++count == num_blocks) {
215  goto cont_blocks_found;
216  }
217  continue;
218  }
219  snode = NULL;
220  } else {
221  pnode = cnode;
222  }
223  }
224  }
225 cont_blocks_found:
226  if (snode) {
227  cnode = get_free_hk_node(index);
228 
229  if(!cnode) {
230  DIE_NOW(0, "Panic: No free house keeping nodes for buddy allocator!\n");
231  }
232 
233  sw_memcpy(cnode, snode, sizeof(struct sw_free_area));
234  while (count) {
235  /*
236  * Latch the next node to be released because
237  * after list_del information in snode will be gone.
238  */
239  pnode =
240  list_entry(snode->head.next,
241  struct sw_free_area, head);
242  list_del(&snode->head);
243  buddy_heap[index].free_area[idx].count--;
244  count--;
245  snode = pnode;
246  }
247  if (buddy_heap[index].free_area[idx].count == 0) {
248  INIT_LIST_HEAD(&buddy_heap[index].free_area[idx].head);
249  }
250 
251  return cnode;
252  }
253  }
254 
255  return NULL;
256 }

265 static struct sw_free_area *buddy_get_block(int index, unsigned int idx)
266 {
267 
268  struct sw_free_area *farea = NULL, *rarea = NULL;
269  struct list *lm;
270  int blk_sz;
271 
272  if (idx >= BINS_MAX_ORDER) {
273  return NULL;
274  }
275 
276  if (list_empty(&buddy_heap[index].free_area[idx].head)) {
277  /*
278  * We borrowed a block from higher order. keep half of it and rest half
279  * give to the caller.
280  */
281  /* This is what we keep for us for further allocation request. */
282  farea = buddy_get_block(index, idx + 1);
283  if (farea) {
284  rarea = get_free_hk_node(index);
285  if (rarea) {
286  blk_sz = MIN_BLOCK_SIZE << idx;
287  list_add_tail(&buddy_heap[index].free_area[idx].head,
288  &farea->head);
289  buddy_heap[index].free_area[idx].count++;
290 
291  /* this is our buddy we will give to caller */
292  rarea->map = farea->map + blk_sz;
293  }
294  }
295  } else {
296  lm = list_pop_tail(&buddy_heap[index].free_area[idx].head);
297  buddy_heap[index].free_area[idx].count--;
298  if (buddy_heap[index].free_area[idx].count == 0) {
299  INIT_LIST_HEAD(&buddy_heap[index].free_area[idx].head);
300  }
301  rarea = list_entry(lm, struct sw_free_area, head);
302  }
303 
304  return rarea;
305 }

367 {
368  struct list *pos;
369  struct sw_free_area *lfa = NULL, *cfa = NULL;
370  void *lmap = NULL;
371 
372 restart:
373  if (bin == BINS_MAX_ORDER - 1) {
374  return 0;
375  }
376 
377  list_for_each(pos, &buddy_heap[index].free_area[bin].head) {
378  cfa = list_entry(pos, struct sw_free_area, head);
379  if (lmap) {
380  if ((lmap + (MIN_BLOCK_SIZE << bin)) == cfa->map) {
381  OTZ_DPRINTK
382  ("Coalescing 0x%x and 0x%x and giving back to bin 0x%x\n",
383  (unsigned int)lfa->map,
384  (unsigned int)cfa->map, bin + 1);
385  list_del(&cfa->head);
386  list_del(&lfa->head);
387  sw_memset(cfa, 0,
388  sizeof(struct sw_free_area));
389  add_free_area_to_bin(index, lfa, bin + 1);
390  lmap = NULL;
391  lfa = NULL;
392  cfa = NULL;
393  buddy_heap[index].free_area[bin].count -= 2;
394 
395  /* restart the list afresh */
396  goto restart;
397  } else {
398  lmap = cfa->map;
399  lfa = cfa;
400  }
401  } else {
402  lmap = cfa->map;
403  lfa = cfa;
404  }
405  }
406 
407  coalesce_buddies(index, bin + 1);
408 
409  return 0;

447 void *sw_buddy_malloc(int index, unsigned int size)
448 {
449  int idx = 0;
450  int curr_blk = MIN_BLOCK_SIZE;
451  struct sw_free_area *farea = NULL;
452  struct sw_alloced_area *aarea = NULL;
453  u32 bneeded;
454 
455  if (size > buddy_heap[index].heap_size) {
456  return NULL;
457  }
458 
459  if (size > MAX_BLOCK_SIZE) {
460  bneeded =
461  (size %
462  MAX_BLOCK_SIZE ? ((size / MAX_BLOCK_SIZE) +
463  1) : size / MAX_BLOCK_SIZE);
464  farea = buddy_get_contiguous_block(index, bneeded, BINS_MAX_ORDER - 1);
465  if (farea) {
466  aarea = get_free_ac_node(index);
467 
468  if(!aarea) {
469  DIE_NOW(0,
470  "Panic: No house keeping node available for buddy allocator.!\n");
471  }
472 
473  aarea->map = farea->map;
474  aarea->blk_sz = MAX_BLOCK_SIZE * bneeded;
475  aarea->bin_num = BINS_MAX_ORDER - 1;
476  list_add_tail(&buddy_heap[index].current.head, &aarea->head);
477  buddy_heap[index].current.count++;
478  free_hk_node(farea);
479  return aarea->map;
480  }
481  }
482 
483  for (idx = 0; idx <= BINS_MAX_ORDER; idx++) {
484  if (size > curr_blk) {
485  curr_blk <<= 1;
486  } else {
487  farea = buddy_get_block(index, idx);
488  if (farea) {
489  aarea = get_free_ac_node(index);
490  if (!aarea) {
491  OTZ_DPRINTK("Bummer! No free alloc node?\n");
492  return NULL;
493  }
494  aarea->map = farea->map;
495  aarea->blk_sz = curr_blk;
496  aarea->bin_num = idx;
497  sw_memset(farea, 0,
498  sizeof(struct sw_free_area));
499  free_hk_node(farea);
500  list_add_tail(&buddy_heap[index].current.head,
501  &aarea->head);
502  buddy_heap[index].current.count++;
503  return aarea->map;
504  }
505  }
506  }
507 
508  return NULL;
509 }
