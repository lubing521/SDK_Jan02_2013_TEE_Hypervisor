86 void init_pl111( u32 clcd_base,
87  s32int width,
88  s32int height,
89  u32 frame_base )
90 {
91  volatile u32* clcd = (volatile u32*) clcd_base;
92 
93  /* PL111 register offsets (32-bit words) */
94  const s32int PL111_TIM0 = (0x00/4);
95  const s32int PL111_TIM1 = (0x04/4);
96  const s32int PL111_TIM2 = (0x08/4);
97  const s32int PL111_TIM3 = (0x0C/4);
98  const s32int PL111_UBAS = (0x10/4);
99  const s32int PL111_LBAS = (0x14/4);
100  const s32int PL111_CNTL = (0x18/4);
101  const s32int PL111_IENB = (0x1C/4);
102 
103  /* Timing number for an 8.4" LCD screen for use on a VGA screen */
104  u32 TIM0_VAL =
105  ( (((width/16)-1)<<2) | (63<<8) | (31<<16) | (63<<8) );
106  u32 TIM1_VAL = ( (height - 1) | (24<<10) | (11<<16) | (9<<24) );
107  u32 TIM2_VAL = ( (0x7<<11) | ((width - 1)<<16) | (1<<26) );
108 
109  /* Statically allocate memory for screen buffer */
110  g_framebuff = (u32*) frame_base;
111 
112  sw_printf("FRAME_BASE:%x\n", g_framebuff);
113 
114  /* Program the CLCD controller registers and start the CLCD */
115  clcd[ PL111_TIM0 ] = TIM0_VAL;
116  clcd[ PL111_TIM1 ] = TIM1_VAL;
117  clcd[ PL111_TIM2 ] = TIM2_VAL;
118  clcd[ PL111_TIM3 ] = 0;
119  clcd[ PL111_UBAS ] = frame_base;
120  clcd[ PL111_LBAS ] = 0;
121  clcd[ PL111_IENB ] = 0;
122 
123  /* Set the control register: 16BPP, Power OFF */
124  if(!rgb444) {
125  /* RGB 565 mode */
126  clcd[ PL111_CNTL ] = (1<<0) | (4<<0) | (1<<5) | (0<<8) | (4 << 1);
127  }
128  else {
129  /* RGB 444 mode */
130  clcd[ PL111_CNTL ] = (1<<0) | (1<<5) | (4<<1);
131  }
132 
133  /* Power ON */
134  clcd[ PL111_CNTL ] |= (1<<11);
135  fb_open = 1;
136 }

140 {
141  const u32 dvi_mux = 0;
142 
143  /* VE System Register 32-bit word offsets */
144  const s32int VE_SYS_ID = (0x00/4);
145  const s32int VE_SYS_CFG_DATA = (0xA0/4);
146  const s32int VE_SYS_CFG_CTRL = (0xA4/4);
147  const s32int VE_SYS_ID_HBI = 0x225;
148 
149  volatile u32* ve_sysreg = (volatile u32*) VE_SYSTEM_REGS;
150 
151  /* Set CLCD clock
152  * SYS_CFG_DATA sets oscillator rate value as 5.4MHz
153  * SYS_CFG_CTRL
154  * ( start=1 | write=1 | function=1 | site=0 | position=0 | device=1 )
155  */
156  ve_sysreg[ VE_SYS_CFG_DATA ] = 5400000;
157  ve_sysreg[ VE_SYS_CFG_CTRL ] = 0x80000000 |
158  (1<<30) | (1<<20) | (0<<16) | (0<<12) | (1<<0);
159 
160  /* Set DVI mux for correct MMB
161  SYS_CFG_CTRL ( start=1 | write=1 |
162  function=7 | site=0 | position=0 | device=0 )
163  */
164  ve_sysreg[ VE_SYS_CFG_DATA ] = dvi_mux;
165  ve_sysreg[ VE_SYS_CFG_CTRL ] = 0x80000000 |
166  (1<<30) | (7<<20) | (0<<16) | (0<<12) | (0<<0);
167 
168  init_pl111(VE_CLCD_BASE, width, height, VE_FRAME_BASE);
