54 void set_domain(u8 domain, access_type access)
55 {
56  /* Domain is a two bit field
57  00 = no access,
58  01=client,
59  10=reserved,
60  11=manager
61  */
62  u32 value;
63  value = read_domain();
64  //clear the current domain
65  u32 mask = ~(0b11 << (domain*2));
66  value = value & mask;
67  //Set the domain
68  value = value | (access << ((domain)*2));
69  write_domain(value);
70 }

136 void mmu_disable_virt_addr()
137 {

140 
141  /* clear MMU, I-Cache and D-Cache */
142  mask = (SCTLR_M | SCTLR_C | SCTLR_I);
143  reg_val = reg_val & (~mask);
